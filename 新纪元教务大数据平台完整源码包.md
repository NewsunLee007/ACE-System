### **2.1 核心算法引擎 (Service层)**

**文件路径**：backend/services/score\_analysis\_service.py

**设计亮点**：引入 pandas 库处理数据，精准实现了李主任指定的 50-20-30 班级 Z 值考核公式，并严格通过 is\_included=1 过滤异常数据。

import pandas as pd  
from sqlalchemy.orm import Session  
from typing import List, Dict, Any

class ScoreAnalysisService:  
    def \_\_init\_\_(self, db\_session: Session):  
        self.db \= db\_session

    def \_get\_layer\_valid\_scores\_df(self, exam\_id: int, layer\_id: int) \-\> pd.DataFrame:  
        sql\_query \= """  
            SELECT s.student\_id, s.class\_name, s.score\_chinese, s.score\_math,   
                   s.score\_english, s.score\_science, s.score\_society, s.total\_score  
            FROM biz\_scores s  
            JOIN biz\_class\_layer\_details d ON s.class\_name \= d.class\_name  
            WHERE s.exam\_id \= :exam\_id   
              AND d.layer\_id \= :layer\_id   
              AND s.is\_included \= 1  
        """  
        df \= pd.read\_sql(sql\_query, self.db.bind, params={"exam\_id": exam\_id, "layer\_id": layer\_id})  
        return df

    def calculate\_layer\_thresholds(self, exam\_id: int, layer\_id: int) \-\> Dict\[str, Any\]:  
        df \= self.\_get\_layer\_valid\_scores\_df(exam\_id, layer\_id)  
        if df.empty:  
            return {"error": "该分层下暂无有效成绩数据"}

        total\_students \= len(df)  
        df\_sorted \= df.sort\_values(by='total\_score', ascending=False).reset\_index(drop=True)

        percentages \= \[0.20, 0.40, 0.60, 0.80\]  
        results \= {}

        for p in percentages:  
            cutoff\_index \= max(1, int(total\_students \* p)) \- 1  
            threshold\_score \= df\_sorted.loc\[cutoff\_index, 'total\_score'\]  
            top\_df \= df\_sorted\[df\_sorted\['total\_score'\] \>= threshold\_score\]  
              
            means \= top\_df\[\['score\_chinese', 'score\_math', 'score\_english',   
                            'score\_science', 'score\_society', 'total\_score'\]\].mean().round(2).to\_dict()  
              
            label \= f"前{int(p\*100)}%"  
            results\[label\] \= {  
                "下限分\_总分": threshold\_score,  
                "群体均分": means,  
                "达标人数": len(top\_df)  
            }  
        return results

    def calculate\_class\_z\_value(self, exam\_id: int, layer\_id: int, target\_class: str) \-\> Dict\[str, float\]:  
        df \= self.\_get\_layer\_valid\_scores\_df(exam\_id, layer\_id)  
        if df.empty:  
            return {"error": "该分层下暂无有效成绩数据"}

        layer\_mean \= df\['total\_score'\].mean()  
        layer\_std \= df\['total\_score'\].std(ddof=0)

        total\_students \= len(df)  
        df\_sorted \= df.sort\_values(by='total\_score', ascending=False).reset\_index(drop=True)  
          
        idx\_20 \= max(1, int(total\_students \* 0.20)) \- 1  
        idx\_80 \= max(1, int(total\_students \* 0.80)) \- 1  
          
        threshold\_20 \= df\_sorted.loc\[idx\_20, 'total\_score'\]  
        threshold\_80 \= df\_sorted.loc\[idx\_80, 'total\_score'\]

        class\_df \= df\[df\['class\_name'\] \== target\_class\]  
        class\_total\_count \= len(class\_df)  
        if class\_total\_count \== 0:  
            return {"error": "该班级无有效成绩"}

        \# 1\. 计算班级标准分  
        class\_mean \= class\_df\['total\_score'\].mean()  
        standard\_score \= (class\_mean \- layer\_mean) / layer\_std if layer\_std else 0.0

        \# 2\. 计算前20%和前80%的比率  
        class\_top20\_count \= len(class\_df\[class\_df\['total\_score'\] \>= threshold\_20\])  
        class\_top80\_count \= len(class\_df\[class\_df\['total\_score'\] \>= threshold\_80\])  
        ratio\_top20 \= class\_top20\_count / class\_total\_count  
        ratio\_top80 \= class\_top80\_count / class\_total\_count

        \# 3\. 新纪元学校专项加权公式  
        final\_z\_value \= (standard\_score \* 0.50) \+ (ratio\_top20 \* 0.20) \+ (ratio\_top80 \* 0.30)

        return {  
            "班级名称": target\_class,  
            "班级有效人数": class\_total\_count,  
            "班级平均分": round(class\_mean, 2),  
            "统计学标准分": round(standard\_score, 4),  
            "前20%贡献率": round(ratio\_top20, 4),  
            "前80%贡献率": round(ratio\_top80, 4),  
            "最终班级Z值": round(final\_z\_value, 4\)  
        }
